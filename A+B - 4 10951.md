# A+B - 4
https://www.acmicpc.net/problem/10951

```java
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		while(scan.hasNextInt()) {
			int A = scan.nextInt();
			int B = scan.nextInt();
			System.out.println(A+B);	
		}
		scan.close();
	}

}
```
